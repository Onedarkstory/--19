using System;

class Program
{
    static void Main()
    {
        const int RowsCount = 3;
        const int ColumnsCount = 3;
        const int SecondRowIndex = 1;
        const int FirstColumnIndex = 0;
        const int InitialProductValue = 1;

        int[,] matrix = {
            { 2, 4, 6 },
            { 1, 3, 5 },
            { 7, 8, 9 }
        };

        Console.WriteLine("Исходная матрица:");

        for (int i = 0; i < RowsCount; i++)
        {
            for (int j = 0; j < ColumnsCount; j++)
            {
                Console.Write(matrix[i, j] + " ");
            }
            Console.WriteLine();
        }

        Console.WriteLine();

        int rowSum = 0;

        for (int j = 0; j < ColumnsCount; j++)
        {
            rowSum += matrix[SecondRowIndex, j];
        }

        int columnProduct = InitialProductValue;

        for (int i = 0; i < RowsCount; i++)
        {
            columnProduct *= matrix[i, FirstColumnIndex];
        }

        Console.WriteLine($"Сумма второй строки: {rowSum}");
        Console.WriteLine($"Произведение первого столбца: {columnProduct}");
    }
}
